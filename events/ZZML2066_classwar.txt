namespace = zzml2066_classwar

# Classicide Clashes
zzml2066_classwar.1 = {
	type = country_event
	placement = root
	hidden = yes
	immediate = {
		every_scope_state = {
			limit = {
				owner = ROOT
				has_decree = action_classicide_landowners
			}
			save_scope_as = state_undergoing_classicide
		}
		hidden_effect = {
			scope:state_undergoing_classicide = {
				add_devastation = 5
			}
			add_radicals_in_state = {
				value = very_large_radicals
				strata = rich
			}
			add_radicals_in_state = {
				value = small_radicals
				pop_type = clergymen
			}
		}
	}
}

coa_undemocratic_monarchy_trigger = {
	coa_monarchy_trigger = yes
	OR = {
		OR = {
			has_law = law_type:law_autocracy
			has_law = law_type:law_oligarchy
			has_law = law_type:law_elder_council
			#has_law = law_type:law_ecclesiarchy
			has_law = law_type:GovPlus_law_no_voting
		}
		has_law = law_type:law_single_party_state
	}
}

coa_autocracy_trigger = {
	OR = {
		AND = {
			has_law = law_type:law_autocracy
			has_law = law_type:GovPlus_law_no_voting
		}
		has_law = law_type:law_single_party_state
	}
	NOT = {has_law = has_law = law_type:law_council_republic}
}

coa_dictatorship_trigger = {
	coa_republic_trigger = yes
	coa_autocracy_trigger = yes
}

coa_oligarchy_trigger = {
	coa_republic_trigger = yes
	has_law = law_type:law_oligarchy
	has_law = law_type:GovPlus_law_no_voting
}

coa_theocracy_trigger = {
	has_law = law_type:law_theocracy
}

coa_multicultural_trigger = {
	has_law = law_type:law_multicultural
}

coa_anarchy_trigger = {
	has_law = law_type:law_anarchy
}

coa_communist_trigger = {
	has_law = law_type:law_council_republic
	NOT = { has_law = law_type:law_anarchy }
}

coa_commune_trigger = {
	law_type:law_council_republic
	has_law = law_type:law_mass_democracy
	NOT = { has_law = law_type:law_anarchy }
}

coa_technocracy_trigger = {
	has_law = law_type:law_technocracy
}

coa_military_trigger = {
	OR = {
		has_government_type = gov_military_dictatorship
		has_government_type = gov_junta
		has_government_type = gov_republic_martial_law
	}
}

		state_region = {
			is_homeland = irish
		}
	}
}

coa_controls_part_of_france = {
	any_scope_state = {
		state_region = {
			is_homeland = french
		}
	}
}

coa_nihilist_trigger = {
	has_law = law_type:law_state_atheism
	NOR = {
		has_law = law_type:law_theocracy
		has_law = law_type:law_monarchy
		has_law = law_type:law_council_republic
		has_law = law_type:law_anarchy
	}
	any_scope_character = {
		has_ideology = ideology:ideology_atheist
		is_ruler = yes
    }
}

		is_secessionist = yes
		is_revolutionary = yes
	}
}

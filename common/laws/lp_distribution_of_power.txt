# 교권 정치	# laws +
law_ecclesiarchy = {
	group = lawgroup_distribution_of_power
	
	icon = "gfx/interface/icons/law_icons/ecclesiarchy.dds"
	
	progressiveness = -100
	
	is_visible = { exists = law_type:law_no_codified_discrimination }

	disallowing_laws = {
		law_freedom_of_conscience
		law_religious_pluralism
		law_total_separation
		law_state_atheism
	}

	on_activate = {
		if = {
			limit = { NOT = { has_law = law_type:GovPlus_law_no_voting } }
			activate_law = law_type:GovPlus_law_no_voting
		}
	}

	modifier = {
		country_legitimacy_govt_total_clout_add = 180
		country_legitimacy_govt_size_add = 2
		country_legitimacy_ideological_incoherence_mult = -0.1
		country_clergymen_pol_str_mult = 1
		country_authority_add = 100
	}
	
	possible_political_movements = {
		law_autocracy
		law_oligarchy
		law_technocracy
		law_single_party_state
		law_presidential_republic
		law_parliamentary_republic
		GovPlus_law_council_system
	}

	pop_support = {
		value = 0
		add = {
			desc = "POP_CLERGYMEN"			
			if = {
				limit = {
					owner = {
						NOT = { has_law = law_type:law_autocracy }
					}
					is_pop_type = clergymen
					has_state_religion = yes
				}
				value = 0.25
			}
		}
	}

	can_impose = { 
		is_subject_of = scope:initiator	
	} 

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}
